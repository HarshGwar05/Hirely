Step 3

Now we need one SQL query.

Append this to the end of screeningService.js

export const getResultDetails = async (
    screeningResumeId
) => {

    const [rows] = await connection.query(
        `
        SELECT

            r.original_name,

            sr.score,
            sr.decision,
            sr.confidence,
            sr.requirement_coverage,
            sr.analysis_json

        FROM ScreeningResult sr

        JOIN ScreeningResume smr
        ON sr.screening_resume_id = smr.screening_resume_id

        JOIN Resume r
        ON smr.resume_id = r.resume_id

        WHERE sr.screening_resume_id = ?

        LIMIT 1
        `,
        [screeningResumeId]
    );

    if (rows.length === 0) {

        return null;

    }

    rows[0].analysis_json =
        JSON.parse(rows[0].analysis_json);

    return rows[0];

};
